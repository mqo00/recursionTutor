binarySearch([1,4,5,7,8,10], 9) == False

def binarySearch(L, n):
if len(L) == 0:
return False
else:
midIdx = len(L) // 2
midElem = L[midIdx]
if n == midElem:
return True
elif n < midElem:
return binarySearch(L[:midIdx], n)
else:
return binarySearch(L[midIdx + 1:], n)
